// packages/arya-executor/src/mood-persistence.ts
export class MoodPersistence {
  async saveMood(agentId: string, mood: Mood, timestamp: number) {
    await db.insert(agentMoodHistory).values({
      agentId,
      mood: mood.type,
      harmony: mood.harmony,
      timestamp
    });
  }

  async loadMoodHistory(agentId: string, window = 100): Mood[] {
    // Load last 100 mood states
    const rows = await db.select().from(agentMoodHistory)
      .where(eq(agentMoodHistory.agentId, agentId))
      .orderBy(desc(agentMoodHistory.timestamp))
      .limit(window);
    
    return rows.map(row => ({
      type: row.mood,
      harmony: row.harmony,
      timestamp: row.timestamp
    }));
  }

  async predictNextMood(agentId: string): Mood {
    // Use LLM to predict next mood based on history
    const history = await this.loadMoodHistory(agentId);
    const prediction = await openai.chat.completions.create({
      model: 'gpt-4',
      messages: [{
        role: 'user',
        content: `Agent mood history: ${JSON.stringify(history)}. What's the likely next mood?`
      }]
    });
    return JSON.parse(prediction.choices[0].message.content);
  }
}
